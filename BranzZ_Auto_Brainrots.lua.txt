local espPool={}
local function clearESP()
    for _,b in pairs(espPool) do pcall(function() b:Destroy() end) end; espPool={}
end

-- LER RARIDADE
local function getRarity(model)
    for _,obj in pairs(model:GetDescendants()) do
        if obj.Name=="CharRarity" then
            if obj:IsA("TextLabel") or obj:IsA("TextBox") then return obj.Text
            elseif obj:IsA("StringValue") then return obj.Value end
        end
    end
    return nil
end

-- PEGAR QUALQUER PART DO MODEL
local function getAnyPart(model)
    if model.PrimaryPart then return model.PrimaryPart end
    for _,p in pairs(model:GetDescendants()) do
        if p:IsA("BasePart") then return p end
    end
    return nil
end

-- =====================================================
-- TP IMEDIATO SEM CONDIÇÃO DE DISTÂNCIA
-- Seta CFrame direto, sem checar distância nenhuma
-- =====================================================
local function teleportNow(part)
    local char = player.Character
    if not char then return end
    local hrp = char:FindFirstChild("HumanoidRootPart")
    if not hrp then return end
    -- TP DIRETO. Sem if. Sem checar distância. Só seta.
    hrp.CFrame = CFrame.new(part.Position.X, part.Position.Y + 3, part.Position.Z)
    task.wait(0.1)
end

local function goSpawn()
    if not spawnCFrame then return end
    local char = player.Character
    if not char then return end
    local hrp = char:FindFirstChild("HumanoidRootPart")
    if hrp then hrp.CFrame = spawnCFrame end
    task.wait(0.1)
end

-- =====================================================
-- FORÇAR PROXIMITY PROMPT SEM PRECISAR ESTAR PERTO
-- =====================================================
local function forcePrompts(model, root)
    local char = player.Character
    if not char then return end
    local hrp = char:FindFirstChild("HumanoidRootPart")
    if not hrp then return end

    for _, obj in pairs(model:GetDescendants()) do
        if obj:IsA("ProximityPrompt") then
            pcall(function()
                local orig = obj.MaxActivationDistance
                -- Abre distância pra 9999 ANTES de qualquer coisa
                obj.MaxActivationDistance = 9999
                obj.Enabled = true

                -- TP na part exata do prompt
                local pp = obj.Parent
                if pp and pp:IsA("BasePart") then
                    hrp.CFrame = CFrame.new(pp.Position.X, pp.Position.Y+3, pp.Position.Z)
                end
                task.wait(0.05)

                -- Dispara
                fireproximityprompt(obj)
                task.wait(0.05)

                -- Se precisar segurar
                if obj.HoldDuration > 0 then
                    holdproximityprompt(obj)
                    task.wait(obj.HoldDuration + 0.1)
                end

                obj.MaxActivationDistance = orig
            end)
        end
    end
end

-- =====================================================
-- LOOP DE COLETA
-- Sem NENHUMA checagem de distância
-- Só lê a pasta, acha o model com a raridade,
-- TP direto, interage, volta pro spawn. Fim.
-- =====================================================
local function collect(targetRarity, espColor, activeCheck)
    local folder = Workspace:FindFirstChild("ClientBrainrotVisuals")
    if not folder then
        setStatus("Pasta nao encontrada!", Color3.fromRGB(255,80,80))
        return
    end

    clearESP()
    local count = 0

    for _, model in pairs(folder:GetChildren()) do
        -- Para o loop se toggle foi desligado
        if not activeCheck() then
            clearESP()
            setStatus("Pronto", Color3.fromRGB(80,220,120))
            return
        end

        if not model or not model.Parent then
            task.wait(0); continue
        end

        local rarity = getRarity(model)

        if rarity and rarity:lower():find(targetRarity:lower()) then
            local root = getAnyPart(model)

            if root and root.Parent then
                -- ESP
                addESP(root, espColor, "⭐ "..rarity)
                setStatus(rarity.." > "..count, espColor)

                -- ===================================
                -- TP DIRETO. ZERO CHECAGEM.
                -- ===================================
                teleportNow(root)

                -- INTERAGE COM PROMPT
                forcePrompts(model, root)

                count += 1
                task.wait(0.4)

                -- VOLTA PRO SPAWN
                goSpawn()
            end
        end

        task.wait(0)
    end

    clearESP()
    setStatus("Coletados: "..count, espColor)
end

-- INFJUMP + WALLHACK
-- =====================================================
local jumpConn = nil

RunService.Heartbeat:Connect(function()
    if not State.InfJump then
        if jumpConn then jumpConn:Disconnect(); jumpConn = nil end
        return
    end

    local char = player.Character
    if not char then return end
    local hum = char:FindFirstChild("Humanoid")
    local hrp = char:FindFirstChild("HumanoidRootPart")
    if not hum or not hrp then return end

    -- Speed e jump power constantes
    hum.WalkSpeed = 140
    hum.JumpPower = 130

    -- Wallhack: sem colisão
    for _, p in pairs(char:GetDescendants()) do
        if p:IsA("BasePart") then p.CanCollide = false end
    end

    -- Pulo infinito: mantém estado jumping enquanto no ar
    local st = hum:GetState()
    if st == Enum.HumanoidStateType.Freefall then
        hum:ChangeState(Enum.HumanoidStateType.Jumping)
    end
end)

-- PULO COM ESPAÇO INFINITO

    local char = player.Character
    if not char then return end
    local hum = char:FindFirstChild("Humanoid")
    if hum then hum:ChangeState(Enum.HumanoidStateType.Jumping) end
end)
